### 2. 修改渲染逻辑

在 `main.js` 文件的 `renderExamContent` 函数中，将第456-459行的代码：

```javascript
html += `<div style="margin-top:8px;">
    <button class="btn btn-sm" onclick="editQuestionInExam(${q.id})"><i class="fas fa-edit"></i> 编辑</button>
    <button class="btn btn-danger btn-sm" onclick="deleteQuestionInExam(${q.id})"><i class="fas fa-trash"></i> 删除</button>
</div>`;
```

替换为：

```javascript
html += `<div style="margin-top:8px;">
    <div class="menu-container">
        <button class="menu-button" onclick="toggleMenu(${q.id})">
            <span>操作</span>
            <span>▼</span>
        </button>
        <div id="menu-${q.id}" class="menu-dropdown hidden">
            <button class="menu-item" onclick="editQuestionInExam(${q.id})")">编辑</button>
            <button class="menu-item delete" onclick="deleteQuestionInExam(${q.id})")">删除</button>
        </div>
    </div>
</div>`;
```

### 3. 添加菜单交互

在 `main.js` 文件中添加以下函数：

```javascript
// 切换菜单显示/隐藏
function toggleMenu(qid) {
    const menu = document.getElementById(`menu-${qid}`);
    if (menu) {
        menu.classList.toggle('hidden');
    }
}

// 点击页面其他地方关闭菜单
document.addEventListener('click', function(event) {
    const menuContainers = document.querySelectorAll('.menu-container');
    menuContainers.forEach(container => {
        if (!container.contains(event.target)) {
            const menu = container.querySelector('.menu-dropdown');
            if (menu) {
                menu.classList.add('hidden');
            }
        }
    });
});
```
